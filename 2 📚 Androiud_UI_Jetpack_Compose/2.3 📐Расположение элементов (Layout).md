import androidx.compose.foundation.lazy.items

import androidx.compose.material3.Text
import androidx.compose.material3.TextField
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier

@Composable
fun DynamicUserList() {
    var users by remember { mutableStateOf(mutableListOf<User>()) }
    var nextUserId by remember { mutableStateOf(1) }
    var newUserName by remember { mutableStateOf("") }

    Column {
        Row(
            verticalAlignment = Alignment.CenterVertically,
            modifier = Modifier.padding(16.dp)
        ) {
            TextField(
                value = newUserName,
                onValueChange = { newUserName = it },
                label = { Text("New user name") },
                modifier = Modifier.weight(1f) // TextField –∑–∞–Ω–∏–º–∞–µ—Ç –≤—Å–µ –¥–æ—Å—Ç—É–ø–Ω–æ–µ –º–µ—Å—Ç–æ –ø–æ —à–∏—Ä–∏–Ω–µ

            Spacer(modifier = Modifier.width(8.dp))
            Button(onClick = {
                if (newUserName.isNotBlank()) {
                    users.add(User(nextUserId++, newUserName))
                    newUserName = "" // –û—á–∏—â–∞–µ–º –ø–æ–ª–µ –≤–≤–æ–¥–∞
                    //–°–æ–∑–¥–∞–µ–º –∫–æ–ø–∏—é —Å–ø–∏—Å–∫–∞ –¥–ª—è –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è —Å–æ—Å—Ç–æ—è–Ω–∏—è
                    users = users.toMutableList()
                }
            }) {
                Text("Add")
            }
        }

        LazyColumn {
            items(users) { user ->
                Row(
                    modifier = Modifier
                        .fillMaxWidth()
                        .padding(16.dp),
                    horizontalArrangement = Arrangement.SpaceBetween //–†–∞—Å–ø—Ä–µ–¥–µ–ª—è–µ–º —ç–ª–µ–º–µ–Ω—Ç—ã –ø–æ —à–∏—Ä–∏–Ω–µ

                    //–°–æ–∑–¥–∞–µ–º –∫–æ–ø–∏—é —Å–ø–∏—Å–∫–∞ –¥–ª—è –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è —Å–æ—Å—Ç–æ—è–Ω–∏—è

                  }) {
                    Text("Delete")
                  }
                }

            }
        }
    }
}
